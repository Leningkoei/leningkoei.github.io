this.minSubArrayLen = function(target, nums) { let sum = 0; let fast = 0; let slow = -1; let result = nums.length; while (fast < nums.length) { sum += nums[fast++]; // 上1行 === 下2行; // sum += nums[fast]; // fast++; if (sum >= target) { while (sum >= target) { sum -= nums[++slow]; // 上1行 === 下2行; // slow++; // sum -= nums[slow]; } result = Math.min(result, fast - slow); } } return slow === -1 ? 0 : result; }
