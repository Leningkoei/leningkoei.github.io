this.generate = function(numRows) { let result = new Array(numRows); result[0] = [ 1 ]; for (let rowIndex = 1; rowIndex < numRows; rowIndex++) { let row = new Array(rowIndex + 1).fill(1); for (let i = 1, j = row.length - 2; i <= j; i++, j--) { row[i] = result[rowIndex - 1][i - 1] + result[rowIndex - 1][i]; row[j] = result[rowIndex - 1][j - 1] + result[rowIndex - 1][j]; } result[rowIndex] = row; } return result; }
